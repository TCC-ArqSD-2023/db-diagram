// Tabelas principais
table associado{
  id int [pk, increment]
  nome varchar
  sobrenome varchar
  data_nascimento date
  estado_civil estado_civil
  sexo sexo
  naturalidade_uf uf
  naturalidade_cidade varchar
  numero_documento varchar
  uf_documento uf
  orgao_documento varchar
  data_documento date
  nome_mae varchar
  nome_pai varchar
  endereco_id int
  email varchar
  telefone varchar
  situacao situacao_associado
}

table prestador{
  id int [pk, increment]
  nome varchar
  sobrenome varchar
  data_nascimento date
  sexo sexo
  numero_documento_profissional varchar
  endereco_atendimento_id int
  especialidade_id int
  telefone varchar
}

table conveniado{
  id int [pk, increment]
  nome varchar
  endereco_id int
  telefone varchar
}

table plano{
  id int [pk, increment]
  nome varchar
  codigo_ans varchar
  categoria categoria_plano
  tipo tipo_plano
  odonto boolean
  idade_min int
  idade_max int
  quantidade_consultas_ano int
  quantidade_exames_ano int
  valor_mensalidade double
}

table consulta{
  id int [pk, increment]
  associado_id int
  disponibilidade_consulta_id int
  situacao situacao_atendimento
}

table disponibilidade_consulta{
  id int [pk, increment]
  prestador_id int
  hora_inicio datetime
  hora_fim datetime
}

table exame{
  id int [pk, increment]
  associado_id int
  disponibilidade_exame_id int
  situacao situacao_atendimento
}

table disponibilidade_exame{
  id int [pk, increment]
  conveniado_id int
  exame_id int
  hora_inicio datetime
  hora_fim datetime
}

table tipo_exame{
  id int [pk, increment]
  nome varchar
}
