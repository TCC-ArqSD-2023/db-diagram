// Tabelas principais
table saf.associado{
  id int [pk, increment]
  nome varchar
  sobrenome varchar
  data_nascimento date
  estado_civil estado_civil
  sexo sexo
  naturalidade_uf uf
  naturalidade_cidade varchar
  numero_documento varchar
  uf_documento uf
  orgao_documento varchar
  data_documento date
  nome_mae varchar
  nome_pai varchar
  endereco_id int
  email varchar
  telefone varchar
  situacao situacao_associado
  plano_id int
}

table saf.prestador{
  id int [pk, increment]
  nome varchar
  sobrenome varchar
  data_nascimento date
  sexo sexo
  numero_documento_profissional varchar
  especialidade_id int
  endereco_atendimento_id int
  telefone varchar
}

table saf.conveniado{
  id int [pk, increment]
  nome varchar
  endereco_id int
  telefone varchar
}

table saf.plano{
  id int [pk, increment]
  nome varchar
  codigo_ans varchar
  categoria categoria_plano
  tipo tipo_plano
  odonto boolean
  idade_min int
  idade_max int
  quantidade_consultas_ano int
  quantidade_exames_ano int
  valor_mensalidade double
}

table sgps.consulta{
  id int [pk, increment]
  prestador_id int
  associado_id int
  hora_inicio datetime
  hora_fim datetime
  situacao situacao_atendimento
}

table sgps.exame{
  id int [pk, increment]
  conveniado_id int
  tipo_exame_id int
  associado_id int
  hora_inicio datetime
  hora_fim datetime
  situacao situacao_atendimento
}

table sgps.tipo_exame{
  id int [pk, increment]
  nome varchar
}

//Tabelas de suporte

table saf.endereco{
  id int [pk, increment]

  numero varchar

  uf uf
